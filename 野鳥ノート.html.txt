let records = JSON.parse(localStorage.getItem('yachoRecords') || '[]');

// ===== RENDER BIRD LIST =====
function renderBirdList(filter = "") {
  const container = document.getElementById('birdList');
  const groups = {};

  BIRDS.forEach(b => {
    if (!b.name.includes(filter) && !b.en.toLowerCase().includes(filter.toLowerCase())) return;
    if (!groups[b.group]) groups[b.group] = [];
    groups[b.group].push(b);
  });

  container.innerHTML = Object.entries(groups).map(([group, birds]) => `
    <div class="bird-group">
      <div class="group-label">${group}</div>
      ${birds.map(b => {
        const isChecked = records.some(r => r.bird === b.name);
        return `
        <div class="bird-item ${isChecked ? 'checked' : ''}" onclick="quickRecord('${b.name}')">
          <div class="bird-check">
            <svg viewBox="0 0 12 12" fill="none">
              <polyline points="2,6 5,9 10,3" stroke="white" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"/>
            </svg>
          </div>
          <div class="bird-info">
            <div class="bird-name-jp">${b.name}</div>
            <div class="bird-name-en">${b.en}</div>
          </div>
          <div class="bird-icon">${b.icon}</div>
        </div>`;
      }).join('')}
    </div>
  `).join('');
}

function filterBirds() {
  renderBirdList(document.getElementById('searchInput').value);
}

// ===== QUICK RECORD FROM LIST =====
function quickRecord(birdName) {
  const today = new Date().toISOString().split('T')[0];
  const record = {
    id: Date.now(),
    bird: birdName,
    date: today,
    place: "",
    memo: ""
  };
  records.unshift(record);
  saveRecords();
  renderBirdList(document.getElementById('searchInput').value);
  renderLog();
  showToast(`${birdName} を記録しました`);
}

// ===== ADD RECORD FROM FORM =====
function addRecord() {
  const bird = document.getElementById('recordBird').value;
  const date = document.getElementById('recordDate').value;
  const place = document.getElementById('recordPlace').value;
  const memo = document.getElementById('recordMemo').value;

  if (!bird) { showToast("鳥を選んでください"); return; }
  if (!date) { showToast("日付を入力してください"); return; }

  const record = { id: Date.now(), bird, date, place, memo };
  records.unshift(record);
  saveRecords();
  renderBirdList();
  renderLog();
  showToast(`${bird} を記録しました！`);

  document.getElementById('recordBird').value = "";
  document.getElementById('recordPlace').value = "";
  document.getElementById('recordMemo').value = "";
  switchTab('log');
}

// ===== RENDER LOG =====
function renderLog() {
  const container = document.getElementById('logList');
  const total = records.length;
  const species = new Set(records.map(r => r.bird)).size;
  document.getElementById('statTotal').textContent = total;
  document.getElementById('statSpecies').textContent = species;

  if (records.length === 0) {
    container.innerHTML = `<div class="log-empty">まだ記録がありません。<br>鳥リストからチェックしてみよう！</div>`;
    return;
  }

  container.innerHTML = records.map(r => `
    <div class="log-item">
      <div class="log-header">
        <div class="log-bird">${r.bird}</div>
        <button class="log-delete" onclick="deleteRecord(${r.id})">✕</button>
      </div>
      <div class="log-meta">
        <span>📅 ${r.date}</span>
        ${r.place ? `<span>📍 ${r.place}</span>` : ''}
        ${r.memo ? `<span>💬 ${r.memo}</span>` : ''}
      </div>
    </div>
  `).join('');
}

function deleteRecord(id) {
  records = records.filter(r => r.id !== id);
  saveRecords();
  renderBirdList(document.getElementById('searchInput').value);
  renderLog();
}

function saveRecords() {
  localStorage.setItem('yachoRecords', JSON.stringify(records));
}

// ===== TABS =====
function switchTab(tab) {
  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  document.getElementById('tab-' + tab).classList.add('active');
  event.target.classList.add('active');
}

// ===== TOAST =====
function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 2200);
}

// ===== INIT =====
function init() {
  // Populate bird select
  const sel = document.getElementById('recordBird');
  const groups = {};
  BIRDS.forEach(b => {
    if (!groups[b.group]) groups[b.group] = [];
    groups[b.group].push(b);
  });
  Object.entries(groups).forEach(([group, birds]) => {
    const og = document.createElement('optgroup');
    og.label = group;
    birds.forEach(b => {
      const opt = document.createElement('option');
      opt.value = b.name;
      opt.textContent = b.name;
      og.appendChild(opt);
    });
    sel.appendChild(og);
  });

  // Set today's date
  document.getElementById('recordDate').value = new Date().toISOString().split('T')[0];

  renderBirdList();
  renderLog();
}

init();
